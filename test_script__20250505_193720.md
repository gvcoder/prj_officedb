from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
import time
import os

# Path to your picture
PICTURE_PATH = os.path.abspath('path_to_picture.jpg')  # Replace with your valid picture path
INVALID_PICTURE_PATH = os.path.abspath('invalid_file.txt')  # Replace with an invalid file path

# Instantiate the browser
driver = webdriver.Chrome()

def visit_practice_form():
    driver.get('https://www.tutorialspoint.com/selenium/practice/selenium_automation_practice.php')
    print("Visited the practice form page")
    time.sleep(2)

def successful_form_submission_valid_data():
    visit_practice_form()
    
    driver.find_element(By.NAME, 'name').send_keys("John Doe")
    print("Filled Name field")
    
    driver.find_element(By.NAME, 'email').send_keys("john.doe@example.com")
    print("Filled Email ID field")
    
    driver.find_element(By.XPATH, "//input[@value='Male']").click()
    print("Selected Male as Gender")
    
    driver.find_element(By.NAME, 'mobile').send_keys("9876543210")
    print("Filled Mobile field")
    
    driver.find_element(By.NAME, 'dob').send_keys("01/01/1990")
    print("Filled Date Of Birth")
    
    driver.find_element(By.NAME, 'subjects').send_keys("Maths")
    print("Filled Subjects")
    
    driver.find_element(By.CSS_SELECTOR, 'input[type="file"]').send_keys(PICTURE_PATH)
    print("Uploaded Picture")
    
    driver.find_element(By.NAME, 'currentaddress').send_keys("123 Main St")
    print("Filled Current Address")
    
    driver.find_element(By.ID, 'dropdown1').send_keys("California")
    print("Selected State")
    
    driver.find_element(By.ID, 'dropdown2').send_keys("Los Angeles")
    print("Selected City")
    
    driver.find_element(By.ID, 'submit').click()
    print("Form submitted successfully")
    
    time.sleep(3)
    
    try:
        confirmation_message = WebDriverWait(driver, 10).until(
            EC.visibility_of_element_located((By.CLASS_NAME, 'confirmation-message'))
        )
        print(confirmation_message.text)
    except Exception as e:
        print("Error fetching confirmation message:", str(e))
