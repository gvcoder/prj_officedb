from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
import time
from selenium.webdriver.common.keys import Keys

class TestAutomationForm:
    def setup_method(self):
        self.driver = webdriver.Chrome()
        self.driver.get("https://www.tutorialspoint.com/selenium/practice/selenium_automation_practice.php")
        print("Navigated to the automation practice form page.")

    def teardown_method(self):
        self.driver.quit()
        print("Closed the browser.")

    def fill_form(self, name, email, gender, mobile, dob, subject, hobbies, address, state, city, picture=None):
        try:
            # Fill Name
            name_field = self.driver.find_element(By.NAME, 'firstname')
            name_field.clear()
            name_field.send_keys(name)
            time.sleep(1)

            # Fill Email
            email_field = self.driver.find_element(By.NAME, 'email')
            email_field.clear()
            email_field.send_keys(email)
            time.sleep(1)

            # Select Gender
            if gender == "Male":
                self.driver.find_element(By.ID, 'gender-male').click()
            else:
                self.driver.find_element(By.ID, 'gender-female').click()
            time.sleep(1)

            # Fill Mobile
            mobile_field = self.driver.find_element(By.NAME, 'mobile')
            mobile_field.clear()
            mobile_field.send_keys(mobile)
            time.sleep(1)

            # Select Date Of Birth
            dob_field = self.driver.find_element(By.ID, 'dob')
            dob_field.clear()
            dob_field.send_keys(dob)
            time.sleep(1)

            # Fill Subjects
            subject_field = self.driver.find_element(By.ID, 'subjects')
            subject_field.clear()
            subject_field.send_keys(subject)
            time.sleep(1)

            # Select Hobbies
            if hobbies:
                self.driver.find_element(By.ID, 'hobby-reading').click()
            time.sleep(1)

            # Upload Picture
            if picture:
                picture_field = self.driver.find_element(By.ID, 'uploadpicture')
                picture_field.send_keys(picture)
                time.sleep(1)

            # Fill Current Address
            address_field = self.driver.find_element(By.NAME, 'address')
            address_field.clear()
            address_field.send_keys(address)
            time.sleep(1)

            # Select State
            state_field = self.driver.find_element(By.NAME, 'state')
            state_field.click()
            state_field.send_keys(state)
            time.sleep(1)
            state_field.send_keys(Keys.RETURN)

            # Select City
            city_field = self.driver.find_element(By.NAME, 'city')
            city_field.click()
            city_field.send_keys(city)
            time.sleep(1)
            city_field.send_keys(Keys.RETURN)

            # Click Submit 
            self.driver.find_element(By.NAME, 'submit').click()
            time.sleep(2)

            print("Form submitted successfully with the name:", name)
            return self.driver.find_element(By.CLASS_NAME, 'success').text

        except Exception as e:
            print("Error occurred while filling the form:", str(e))
            return "Form submission failed due to an error."

    def test_successful_submission(self):
        self.setup_method()
        result = self.fill_form("John Doe", "john.doe@example.com", "Male", "9876543210", "01/01/1990", 
                                "Mathematics", "Reading", "123 Elm St, Springfield", "NCR", "Delhi", 
                                "path/to/your/picture.jpg")
        assert "success" in result.lower()
        self.teardown_method()

    def test_empty_name_submission(self):
        self.setup_method()
        result = self.fill_form("", "john.doe@example.com", "Male", "9876543210", "01/01/1990", 
                                "Mathematics", "Reading", "123 Elm St, Springfield", "NCR", "Delhi")
        assert "error" in result.lower()
        self.teardown_method()

    def test_invalid_email_submission(self):
        self.setup_method()
        result = self.fill_form("John Doe", "john.doe", "Male", "9876543210", "01/01/1990", 
                                "Mathematics", "Reading", "123 Elm St, Springfield", "NCR", "Delhi")
        assert "error" in result.lower()
        self.teardown_method()

    def test_short_mobile_number_submission(self):
        self.setup_method()
        result = self.fill_form("John Doe", "john.doe@example.com", "Male", "12345", "01/01/1990", 
                                "Mathematics", "Reading", "123 Elm St, Springfield", "NCR", "Delhi")
        assert "error" in result.lower()
        self.teardown_method()

    def test_maximum_length_submission(self):
        self.setup_method()
        long_name = "a" * 100  # assuming 100 is the max length
        long_subject = "b" * 100  # assuming 100 is the max length
        result = self.fill_form(long_name, "john.doe@example.com", "Male", "9876543210", "01/01/1990", 
                                long_subject, "Reading", "123 Elm St, Springfield", "NCR", "Delhi", 
                                "path/to/your/picture.jpg")
        assert "success" in result.lower()
        self.teardown_method()

    def test_special_characters_in_name(self):
        self.setup_method()
        result = self.fill_form("John@Doe!", "john.doe@example.com", "Male", "9876543210", "01/01/1990", 
                                "Mathematics", "Reading", "123 Elm St, Springfield", "NCR", "Delhi")
        assert "error" in result.lower()
        self.teardown_method()

    def test_missing_picture_submission(self):

                                "Mathematics", "Reading", "123 Elm St, Springfield", "NCR", "Delhi")
        assert "error" in result.lower()
        self.teardown_method()
